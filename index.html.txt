    {
        "imports": {
            "react": "https://esm.sh/react@18.2.0",
            "react-dom/client": "https://esm.sh/react-dom@18.2.0/client",
            "lucide-react": "https://esm.sh/lucide-react@0.292.0",
            "firebase/app": "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js",
            "firebase/auth": "https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js",

    <script type="text/babel" data-type="module">
        import React, { useState, useEffect, useRef, useMemo } from 'react';
        import { createRoot } from 'react-dom/client';
        import { 
          Mic, MicOff, Plus, Minus, Edit2, Trash2, Download, 
          Sprout, X, AlertCircle, CheckCircle2, Info, MapPin, 
          Camera, Image as ImageIcon, FileText, HelpCircle,
          Ban, LeafyGreen, List, Eye
        } from 'lucide-react';
        
        import { initializeApp } from 'firebase/app';
        import { getAuth, signInAnonymously, onAuthStateChanged } from 'firebase/auth';
        import { getFirestore, collection, doc, setDoc, deleteDoc, onSnapshot } from 'firebase/firestore';

        // ======================================================================
        // PASTE YOUR FIREBASE CONFIG HERE:
        // Replace this entire firebaseConfig object with the one Firebase gives you!
        // ======================================================================
        const firebaseConfig = {
  apiKey: "AIzaSyCJob-YrKAmXpF0zkdrh4t7R__k9M3CiFw",
  authDomain: "dahlia-farm-pro.firebaseapp.com",
  projectId: "dahlia-farm-pro",
  storageBucket: "dahlia-farm-pro.firebasestorage.app",
  messagingSenderId: "350375032767",
  appId: "1:350375032767:web:ca63c5cf7e0a64d616530a"

        };
        // ======================================================================

        let app, auth, db;
        try {
            app = initializeApp(firebaseConfig);
            auth = getAuth(app);
            db = getFirestore(app);
        } catch (error) {
            console.error("Firebase initialization failed:", error);
        }

        function App() {
            const [user, setUser] = useState(null);
            const [plants, setPlants] = useState([]);
            const [isLoading, setIsLoading] = useState(true);
            
            const [currentRow, setCurrentRow] = useState(1);
            const [currentPosition, setCurrentPosition] = useState(1);
            const [viewMode, setViewMode] = useState('current');

            const [isListening, setIsListening] = useState(false);
            const [feedback, setFeedback] = useState({ message: '', type: '' });
            const [speechSupported, setSpeechSupported] = useState(true);
            const [editingPlant, setEditingPlant] = useState(null);
            const [isManualModalOpen, setIsManualModalOpen] = useState(false);
            const [showCommandsHelp, setShowCommandsHelp] = useState(false);
            const [viewingPhoto, setViewingPhoto] = useState(null);
            const [isCompressing, setIsCompressing] = useState(false);
            
            const recognitionRef = useRef(null);
            const listEndRef = useRef(null);

            useEffect(() => {
                const timer = setTimeout(() => setIsLoading(false), 4000);
                return () => clearTimeout(timer);
            }, []);

            useEffect(() => {
                if (viewMode === 'current' && listEndRef.current) {
                    listEndRef.current.scrollIntoView({ behavior: 'smooth' });
                }
            }, [plants, currentPosition, viewMode]);

            // Auth
            useEffect(() => {
                if (!auth) {
                    setIsLoading(false);
                    return;
                }
                signInAnonymously(auth).catch((error) => {
                    console.error("Auth error:", error);
                    setIsLoading(false); 
                });
                
                const unsubscribe = onAuthStateChanged(auth, (currentUser) => {
                    setUser(currentUser);
                    if (!currentUser) setIsLoading(false);
                });
                return () => unsubscribe();
            }, []);

            // Data Sync
            useEffect(() => {
                if (!user || !db) return;
                // Saving to a root collection called 'plants' for the farm
                const plantsRef = collection(db, 'plants');
                
                const unsubscribe = onSnapshot(plantsRef, (snapshot) => {
                    const loadedPlants = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                    setPlants(loadedPlants);
                    setIsLoading(false);
                }, (error) => {
                    setFeedback({ message: 'Offline mode active. Data saving locally.', type: 'info' });
                    setIsLoading(false);
                });
                return () => unsubscribe();
            }, [user]);

            const normalizeTranscript = (text) => {
                const wordToNum = {
                    'one': 1, 'two': 2, 'to': 2, 'too': 2, 'three': 3, 'four': 4, 'for': 4,
                    'five': 5, 'six': 6, 'seven': 7, 'eight': 8, 'nine': 9, 'ten': 10,
                    'eleven': 11, 'twelve': 12, 'thirteen': 13, 'fourteen': 14, 'fifteen': 15,
                    'sixteen': 16, 'seventeen': 17, 'eighteen': 18, 'nineteen': 19, 'twenty': 20
                };
                return text.toLowerCase().split(' ').map(word => wordToNum[word] !== undefined ? wordToNum[word] : word).join(' ');
            };

            // Speech Rec
            useEffect(() => {
                const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
                if (!SpeechRecognition) {
                    setSpeechSupported(false);
                    return;
                }

                const recognition = new SpeechRecognition();
                recognition.continuous = false; 
                recognition.interimResults = false;
                recognition.lang = 'en-US';

                recognition.onresult = (event) => {
                    const lastResultIndex = event.results.length - 1;
                    const transcript = event.results[lastResultIndex][0].transcript.trim();
                    processTranscript(transcript);
                };

                recognition.onerror = (event) => {
                    if (event.error !== 'no-speech') {
                        showFeedback(`Microphone error: ${event.error}`, 'error');
                        forceStopMic();
                    }
                };

                recognition.onend = () => setIsListening(false);
                recognitionRef.current = recognition;

                return () => {
                    if (recognitionRef.current) recognitionRef.current.stop();
                };
            }, [currentRow, currentPosition, plants]);

            const forceStopMic = () => {
                if (recognitionRef.current) recognitionRef.current.stop();
                setIsListening(false);
            };

            const showFeedback = (message, type) => {
                setFeedback({ message, type });
                setTimeout(() => setFeedback({ message: '', type: '' }), 4000);
            };

            const toggleListening = () => {
                if (!speechSupported) {
                    showFeedback('Voice not supported on this browser.', 'error');
                    return;
                }
                if (isListening) {
                    forceStopMic();
                    showFeedback('Voice tracking paused.', 'info');
                } else {
                    try {
                        recognitionRef.current.start();
                        setIsListening(true);
                        showFeedback('Listening...', 'info');
                    } catch (e) {
                        console.error(e);
                    }
                }
            };

            const handleRowChange = (newRow) => {
                if (newRow < 1) return;
                setCurrentRow(newRow);
                const plantsInRow = plants.filter(p => p.row === newRow);
                const maxPos = plantsInRow.length > 0 ? Math.max(...plantsInRow.map(p => p.position)) : 0;
                setCurrentPosition(maxPos + 1);
            };

            const processTranscript = (rawText) => {
                const normalizedText = normalizeTranscript(rawText).trim();

                if (normalizedText.match(/^(next|advance)\s*row$/)) {
                    handleRowChange(currentRow + 1);
                    showFeedback(`Moved to Row ${currentRow + 1}`, 'info');
                    return forceStopMic();
                }

                const rowMatch = normalizedText.match(/^(?:go to|jump to|move to|change to|set)?\s*row\s*(\d+)$/);
                if (rowMatch) {
                    const r = parseInt(rowMatch[1], 10);
                    handleRowChange(r);
                    showFeedback(`Moved to Row ${r}`, 'info');
                    return forceStopMic();
                }

                const posMatch = normalizedText.match(/^(?:go to|jump to|move to|change to|set)?\s*(?:plant|position|number|num)\s*(\d+)$/);
                if (posMatch) {
                    const p = parseInt(posMatch[1], 10);
                    setCurrentPosition(p);
                    showFeedback(`Jumped to Plant ${p}`, 'info');
                    return forceStopMic();
                }

                if (normalizedText.match(/^(skip space|empty space|leave empty|mark empty)$/)) {
                    savePlantToCloud({ row: currentRow, position: currentPosition, variety: 'Empty Space', notes: '', photo: null, status: 'skipped' });
                    showFeedback(`Marked position ${currentPosition} as Empty Space`, 'success');
                    setCurrentPosition(prev => prev + 1);
                    return forceStopMic();
                }

                if (normalizedText.match(/^(did not grow|didn't grow|failed|dead|mark failed)$/)) {
                    const existingPlant = plants.find(p => p.row === currentRow && p.position === currentPosition);
                    if (existingPlant) {
                        savePlantToCloud({ ...existingPlant, status: 'failed' });
                        showFeedback(`Marked ${existingPlant.variety} as Failed.`, 'info');
                        setCurrentPosition(prev => prev + 1);
                    } else {
                        showFeedback(`No plant found at Plant ${currentPosition} to mark as failed.`, 'error');
                    }
                    return forceStopMic();
                }

                const variety = rawText.replace(/^(is|called|variety)\s+/i, '').trim().replace(/\b\w/g, l => l.toUpperCase());

                if (variety) {
                    savePlantToCloud({ row: currentRow, position: currentPosition, variety, notes: '', photo: null, status: 'active' });
                    showFeedback(`Added: ${variety}`, 'success');
                    setCurrentPosition(prev => prev + 1);
                    forceStopMic(); 
                } else {
                    showFeedback(`Didn't catch the variety name.`, 'error');
                    forceStopMic();
                }
            };

            const savePlantToCloud = async (plantData) => {
                const existingPlant = plants.find(p => p.row === plantData.row && p.position === plantData.position);
                const idToUse = plantData.id || (existingPlant ? existingPlant.id : Date.now().toString());
                const dataToSave = existingPlant ? { ...existingPlant, ...plantData, id: idToUse } : { ...plantData, id: idToUse };
                
                if (!dataToSave.status) dataToSave.status = 'active';

                setPlants(prev => {
                    const newPlants = [...prev];
                    const idx = newPlants.findIndex(p => p.id === idToUse);
                    if (idx >= 0) newPlants[idx] = dataToSave;
                    else newPlants.push(dataToSave);
                    return newPlants;
                });

                if (!user || !db) return; 
                
                try {
                    const docRef = doc(db, 'plants', idToUse);
                    await setDoc(docRef, dataToSave);
                } catch (err) {
                    console.error("Error saving:", err);
                }
            };

            const deletePlant = async (id) => {
                setPlants(prev => prev.filter(p => p.id !== id));
                if (!user || !db) return;
                try {
                    await deleteDoc(doc(db, 'plants', id));
                } catch (err) {
                    console.error("Error deleting:", err);
                }
            };

            const handlePhotoCapture = async (e, plant) => {
                const file = e.target.files[0];
                if (!file) return;

                setIsCompressing(true);
                showFeedback('Optimizing photo...', 'info');

                try {
                    const compressedDataUrl = await compressImage(file);
                    await savePlantToCloud({ ...plant, photo: compressedDataUrl });
                    showFeedback('Photo attached successfully!', 'success');
                } catch (err) {
                    showFeedback('Failed to process photo.', 'error');
                } finally {
                    setIsCompressing(false);
                    e.target.value = ''; 
                }
            };

            const compressImage = (file) => {
                return new Promise((resolve, reject) => {
                    const reader = new FileReader();
                    reader.onload = (event) => {
                        const img = new Image();
                        img.onload = () => {
                            const canvas = document.createElement('canvas');
                            const MAX_WIDTH = 800;
                            const scaleSize = MAX_WIDTH / img.width;
                            canvas.width = MAX_WIDTH;
                            canvas.height = img.height * scaleSize;
                            const ctx = canvas.getContext('2d');
                            ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
                            resolve(canvas.toDataURL('image/jpeg', 0.6));
                        };
                        img.onerror = reject;
                        img.src = event.target.result;
                    };
                    reader.onerror = reject;
                    reader.readAsDataURL(file);
                });
            };

            const exportToCSV = () => {
                if (plants.length === 0) return alert("No data to export!");
                
                const sortedPlants = [...plants].sort((a, b) => a.row !== b.row ? a.row - b.row : a.position - b.position);

                const headers = ['Row', 'Position', 'Variety', 'Status', 'Notes', 'Has Photo', 'Date Logged'];
                const rows = sortedPlants.map(p => [
                    p.row, 
                    p.position, 
                    `"${p.variety || ''}"`, 
                    p.status || 'active',
                    `"${(p.notes || '').replace(/"/g, '""')}"`, 
                    p.photo ? 'Yes' : 'No',
                    new Date(parseInt(p.id)).toLocaleDateString()
                ]);

                const csvContent = "data:text/csv;charset=utf-8," + [headers.join(","), ...rows.map(e => e.join(","))].join("\n");
                const encodedUri = encodeURI(csvContent);
                const link = document.createElement("a");
                link.setAttribute("href", encodedUri);
                link.setAttribute("download", `dahlia_farm_log_${new Date().toISOString().split('T')[0]}.csv`);
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
            };

            const groupedPlants = useMemo(() => {
                const groups = plants.reduce((acc, plant) => {
                    if (!acc[plant.row]) acc[plant.row] = [];
                    acc[plant.row].push(plant);
                    return acc;
                }, {});
                Object.keys(groups).forEach(row => groups[row].sort((a, b) => a.position - b.position));
                return groups;
            }, [plants]);

            const rowsToRender = viewMode === 'current' 
                ? (groupedPlants[currentRow] ? [currentRow.toString()] : []) 
                : Object.keys(groupedPlants).sort((a,b) => Number(a) - Number(b));

            if (isLoading) {
                return (
                    <div className="min-h-screen bg-[#f4f1ea] flex items-center justify-center font-sans text-emerald-800">
                        <div className="animate-pulse flex flex-col items-center">
                            <Sprout className="w-12 h-12 mb-4" />
                            <p className="font-medium">Connecting to Farm Database...</p>
                        </div>
                    </div>
                );
            }

            return (
                <div className="min-h-screen bg-[#f4f1ea] font-sans text-slate-800 flex flex-col overflow-hidden">
                    
                    <header className="bg-emerald-800 text-white p-3 shadow-md z-20 shrink-0">
                        <div className="max-w-4xl mx-auto flex justify-between items-center">
                            <div className="flex items-center gap-2">
                                <Sprout className="w-6 h-6 text-emerald-300" />
                                <h1 className="text-lg font-bold tracking-wide">Dahlia Pro</h1>
                                {user ? (
                                    <span className="text-[9px] uppercase bg-emerald-900 px-1.5 py-0.5 rounded-full tracking-widest ml-1 flex items-center gap-1">
                                        <span className="w-1.5 h-1.5 rounded-full bg-emerald-400"></span> Sync
                                    </span>
                                ) : (
                                    <span className="text-[9px] uppercase bg-slate-600 px-1.5 py-0.5 rounded-full tracking-widest ml-1 flex items-center gap-1">
                                        <span className="w-1.5 h-1.5 rounded-full bg-slate-400"></span> Local
                                    </span>
                                )}
                            </div>
                            
                            <div className="flex items-center gap-2">
                                <button onClick={() => setViewMode(viewMode === 'current' ? 'all' : 'current')} className="flex items-center gap-1 bg-emerald-900/50 hover:bg-emerald-700 px-2 py-1.5 rounded-lg text-xs transition-colors">
                                    {viewMode === 'current' ? <List className="w-4 h-4" /> : <Eye className="w-4 h-4" />}
                                    <span className="hidden sm:inline">{viewMode === 'current' ? 'View All' : 'Focus Row'}</span>
                                </button>
                                <button onClick={exportToCSV} className="flex items-center gap-1 bg-emerald-700 hover:bg-emerald-600 px-2 py-1.5 rounded-lg text-xs transition-colors">
                                    <Download className="w-4 h-4" />
                                    <span className="hidden sm:inline">Export</span>
                                </button>
                            </div>
                        </div>
                    </header>

                    <main className="flex-1 overflow-y-auto w-full max-w-4xl mx-auto p-4 pb-36">
                        <div className="flex justify-between items-center mb-4">
                            <button onClick={() => setShowCommandsHelp(!showCommandsHelp)} className="text-emerald-700 text-sm font-medium flex items-center gap-1 bg-emerald-100 px-3 py-1.5 rounded-full">
                                <HelpCircle className="w-4 h-4" /> Voice Commands
                            </button>
                            <button onClick={() => { setEditingPlant(null); setIsManualModalOpen(true); }} className="text-sm flex items-center gap-1 bg-slate-800 text-white px-3 py-1.5 rounded-full shadow-sm hover:bg-slate-700 transition">
                                <Plus className="w-4 h-4" /> Manual Entry
                            </button>
                        </div>

                        {showCommandsHelp && (
                            <div className="bg-emerald-50 rounded-xl p-4 mb-6 border border-emerald-100 text-sm animate-in fade-in slide-in-from-top-2">
                                <h3 className="font-bold text-emerald-900 mb-2 border-b border-emerald-200 pb-1">Voice Commands:</h3>
                                <ul className="space-y-1.5 text-emerald-800">
                                    <li><strong className="text-emerald-900">"[Variety]"</strong> - Logs plant & turns off mic</li>
                                    <li><strong className="text-emerald-900">"Empty Space"</strong> - Logs a physical gap</li>
                                    <li><strong className="text-emerald-900">"Did not grow"</strong> - Marks selected plant as failed</li>
                                    <li><strong className="text-emerald-900">"Row 4"</strong> or <strong className="text-emerald-900">"Next row"</strong> - Moves row tracker</li>
                                    <li><strong className="text-emerald-900">"Plant 12"</strong> - Jumps to specific plant</li>
                                </ul>
                            </div>
                        )}

                        <section className="space-y-6">
                            {rowsToRender.length === 0 ? (
                                <div className="text-center py-12 bg-white rounded-2xl border border-dashed border-emerald-200 text-emerald-600/60 mt-4">
                                    <Sprout className="w-12 h-12 mx-auto mb-3 opacity-50" />
                                    <p className="font-medium text-emerald-800">Row {currentRow} is empty.</p>
                                    <p className="text-sm">Tap the mic below to log your first tuber here.</p>
                                </div>
                            ) : (
                                rowsToRender.map(rowNum => (
                                    <div key={rowNum} className="bg-white rounded-2xl shadow-sm border border-emerald-100 overflow-hidden">
                                        <div className="bg-emerald-50 px-4 py-3 border-b border-emerald-100 flex justify-between items-center sticky top-0 z-10 shadow-sm">
                                            <h3 className="font-bold text-emerald-800 text-lg flex items-center gap-2">Row {rowNum}</h3>
                                            <span className="text-xs font-medium bg-white/80 text-emerald-800 px-2 py-1 rounded-full">{groupedPlants[rowNum].length} entries</span>
                                        </div>
                                        
                                        <div className="divide-y divide-slate-100">
                                            {groupedPlants[rowNum].map(plant => {
                                                const isSkipped = plant.status === 'skipped';
                                                const isFailed = plant.status === 'failed';
                                                const isJustAdded = Number(rowNum) === currentRow && plant.position === currentPosition - 1;
                                                
                                                return (
                                                    <div key={plant.id} className={`p-4 flex flex-col sm:flex-row sm:items-center justify-between transition-colors gap-4 
                                                        ${isSkipped ? 'bg-slate-50 opacity-70 border-l-4 border-slate-300' : 
                                                        isFailed ? 'bg-red-50/50 hover:bg-red-50 border-l-4 border-red-300' : 
                                                        isJustAdded ? 'bg-emerald-50/50 border-l-4 border-emerald-400' :
                                                        'hover:bg-slate-50 border-l-4 border-transparent'}
                                                    `}>
                                                        
                                                        <div className="flex items-start sm:items-center gap-4 flex-1">
                                                            <div className={`w-10 h-10 rounded-full flex items-center justify-center font-bold shrink-0 mt-1 sm:mt-0 
                                                                ${isSkipped ? 'bg-slate-200 text-slate-500' :
                                                                isFailed ? 'bg-red-100 text-red-700' :
                                                                isJustAdded ? 'bg-emerald-600 text-white shadow-md' :
                                                                'bg-emerald-100 text-emerald-700'}`}
                                                            >
                                                                #{plant.position}
                                                            </div>
                                                            
                                                            <div className="flex-1">
                                                                <div className="flex items-center gap-2">
                                                                    {isSkipped ? <Ban className="w-4 h-4 text-slate-400" /> : 
                                                                    isFailed ? <AlertCircle className="w-4 h-4 text-red-500" /> : 
                                                                    <LeafyGreen className="w-4 h-4 text-emerald-500" />}
                                                                    
                                                                    <p className={`font-semibold text-lg ${isSkipped ? 'text-slate-500 italic' : isFailed ? 'text-red-900 line-through' : 'text-slate-800'}`}>
                                                                        {plant.variety}
                                                                    </p>
                                                                    
                                                                    {plant.photo && (
                                                                        <button onClick={() => setViewingPhoto(plant.photo)} className="text-emerald-600 hover:text-emerald-700 bg-emerald-50 p-1 rounded-md transition-colors" title="View Photo">
                                                                            <ImageIcon className="w-4 h-4" />
                                                                        </button>
                                                                    )}
                                                                </div>
                                                                
                                                                {isFailed && <p className="text-xs text-red-600 font-medium mt-0.5">Did not grow</p>}
                                                                
                                                                {plant.notes && (
                                                                    <div className="flex items-start gap-1.5 mt-1.5 text-sm text-slate-600 bg-white p-2.5 rounded-lg border border-slate-200 shadow-sm">
                                                                        <FileText className="w-4 h-4 mt-0.5 shrink-0 text-slate-400" />
                                                                        <p className="leading-snug">{plant.notes}</p>
                                                                    </div>
                                                                )}
                                                            </div>
                                                        </div>

                                                        <div className="flex items-center gap-1 self-end sm:self-auto shrink-0 bg-white sm:bg-transparent p-1 sm:p-0 rounded-lg shadow-sm sm:shadow-none border border-slate-100 sm:border-none">
                                                            
                                                            <div className={`relative flex items-center gap-1 p-2 rounded-lg transition overflow-hidden cursor-pointer ${isSkipped ? 'opacity-50 cursor-not-allowed' : 'text-emerald-600 hover:bg-emerald-50'}`} title={isSkipped ? "Cannot add photo to skipped space" : "Add Photo"}>
                                                                <Camera className="w-5 h-5 relative z-10" />
                                                                <span className="text-sm font-medium sm:hidden relative z-10">Photo</span>
                                                                
                                                                {/* NATIVE CAMERA INPUT FOR PRODUCTION */}
                                                                <input 
                                                                    type="file" 
                                                                    accept="image/*" 
                                                                    capture="environment" 
                                                                    className="absolute inset-0 w-full h-full opacity-0 cursor-pointer z-20" 
                                                                    disabled={isSkipped} 
                                                                    onChange={(e) => handlePhotoCapture(e, plant)} 
                                                                />
                                                            </div>

                                                            <div className="w-px h-6 bg-slate-200 mx-1 hidden sm:block"></div>
                                                            <button onClick={() => { setEditingPlant(plant); setIsManualModalOpen(true); }} className="p-2 text-slate-500 hover:text-blue-600 hover:bg-blue-50 rounded-lg transition"><Edit2 className="w-5 h-5" /></button>
                                                            <button onClick={() => deletePlant(plant.id)} className="p-2 text-slate-500 hover:text-red-600 hover:bg-red-50 rounded-lg transition"><Trash2 className="w-5 h-5" /></button>
                                                        </div>
                                                    </div>
                                                );
                                            })}
                                        </div>
                                    </div>
                                ))
                            )}
                            <div ref={listEndRef} className="h-4"></div>
                        </section>
                    </main>

                    <div className="fixed bottom-0 left-0 right-0 bg-white border-t border-slate-200 shadow-[0_-10px_40px_rgba(0,0,0,0.08)] z-40 pb-6 pt-3 px-4 sm:pb-4">
                        <div className="absolute left-0 right-0 -top-14 flex justify-center pointer-events-none">
                            {feedback.message && (
                                <div className={`flex items-center gap-2 px-4 py-2 rounded-full text-sm font-medium shadow-lg animate-in fade-in slide-in-from-bottom-2 ${
                                    feedback.type === 'success' ? 'bg-emerald-600 text-white' :
                                    feedback.type === 'error' ? 'bg-red-600 text-white' : 'bg-slate-800 text-white'
                                }`}>
                                    {feedback.type === 'success' && <CheckCircle2 className="w-4 h-4 shrink-0" />}
                                    {feedback.type === 'error' && <AlertCircle className="w-4 h-4 shrink-0" />}
                                    {feedback.type === 'info' && <Info className="w-4 h-4 shrink-0" />}
                                    {feedback.message}
                                </div>
                            )}
                        </div>

                        <div className="max-w-md mx-auto relative flex justify-between items-center h-16">
                            <div className="flex flex-col items-center w-24">
                                <span className="text-[10px] font-bold text-slate-400 uppercase tracking-wider mb-1">Row</span>
                                <div className="flex items-center gap-2">
                                    <button onClick={() => handleRowChange(currentRow - 1)} className="p-1.5 text-slate-500 hover:bg-slate-100 rounded-full"><Minus className="w-5 h-5"/></button>
                                    <span className="text-2xl font-black text-slate-800 w-8 text-center">{currentRow}</span>
                                    <button onClick={() => handleRowChange(currentRow + 1)} className="p-1.5 text-slate-500 hover:bg-slate-100 rounded-full"><Plus className="w-5 h-5"/></button>
                                </div>
                            </div>

                            <div className="absolute left-1/2 -translate-x-1/2 -top-10">
                                <button
                                    onClick={toggleListening}
                                    disabled={!speechSupported || isCompressing}
                                    className={`flex items-center justify-center w-20 h-20 rounded-full shadow-2xl border-[6px] border-white transition-all duration-300 ${
                                        isListening 
                                        ? 'bg-red-500 text-white animate-pulse scale-105 shadow-[0_0_20px_rgba(239,68,68,0.4)]' 
                                        : 'bg-emerald-600 text-white hover:bg-emerald-500 hover:scale-105'
                                    } ${(!speechSupported || isCompressing) && 'opacity-50 cursor-not-allowed'}`}
                                >
                                    {isListening ? <Mic className="w-8 h-8" /> : <MicOff className="w-8 h-8" />}
                                </button>
                                <div className={`text-center mt-1 text-[10px] font-bold tracking-wider uppercase ${isListening ? 'text-red-500' : 'text-slate-400'}`}>
                                    {isListening ? 'Listening' : 'Tap to Log'}
                                </div>
                            </div>

                            <div className="flex flex-col items-center w-24">
                                <span className="text-[10px] font-bold text-slate-400 uppercase tracking-wider mb-1">Next Plant</span>
                                <div className="flex items-center gap-2">
                                    <button onClick={() => setCurrentPosition(Math.max(1, currentPosition - 1))} className="p-1.5 text-slate-500 hover:bg-slate-100 rounded-full"><Minus className="w-5 h-5"/></button>
                                    <span className="text-2xl font-black text-slate-800 w-8 text-center">{currentPosition}</span>
                                    <button onClick={() => setCurrentPosition(currentPosition + 1)} className="p-1.5 text-slate-500 hover:bg-slate-100 rounded-full"><Plus className="w-5 h-5"/></button>
                                </div>
                            </div>
                        </div>
                    </div>

                    {isManualModalOpen && (
                        <div className="fixed inset-0 bg-slate-900/50 backdrop-blur-sm flex items-center justify-center p-4 z-50">
                            <div className="bg-white rounded-2xl w-full max-w-md shadow-xl overflow-hidden animate-in zoom-in-95">
                                <div className="flex justify-between items-center p-4 border-b">
                                    <h3 className="font-bold text-lg">{editingPlant ? 'Edit Plant Details' : 'Manual Entry'}</h3>
                                    <button onClick={() => setIsManualModalOpen(false)} className="text-slate-400 hover:text-slate-600"><X className="w-5 h-5" /></button>
                                </div>
                                
                                <form 
                                    className="p-4 space-y-4"
                                    onSubmit={(e) => {
                                        e.preventDefault();
                                        const formData = new FormData(e.target);
                                        const r = parseInt(formData.get('row'), 10);
                                        const p = parseInt(formData.get('position'), 10);
                                        
                                        savePlantToCloud({
                                            row: r,
                                            position: p,
                                            variety: formData.get('variety'),
                                            status: formData.get('status'),
                                            notes: formData.get('notes'),
                                            photo: editingPlant?.photo || null,
                                            id: editingPlant?.id || null
                                        });
                                        
                                        if (!editingPlant && r === currentRow && p === currentPosition) {
                                            setCurrentPosition(currentPosition + 1);
                                        }
                                        setIsManualModalOpen(false);
                                    }}
                                >
                                    <div className="grid grid-cols-2 gap-4">
                                        <div className="space-y-1">
                                            <label className="text-sm font-medium text-slate-700">Row</label>
                                            <input type="number" name="row" required min="1" defaultValue={editingPlant?.row || currentRow} className="w-full border border-slate-200 rounded-lg p-2.5 focus:ring-2 focus:ring-emerald-500 outline-none" />
                                        </div>
                                        <div className="space-y-1">
                                            <label className="text-sm font-medium text-slate-700">Position</label>
                                            <input type="number" name="position" required min="1" defaultValue={editingPlant?.position || currentPosition} className="w-full border border-slate-200 rounded-lg p-2.5 focus:ring-2 focus:ring-emerald-500 outline-none" />
                                        </div>
                                    </div>
                                    
                                    <div className="space-y-1">
                                        <label className="text-sm font-medium text-slate-700">Variety Name</label>
                                        <input type="text" name="variety" required defaultValue={editingPlant?.variety || ''} className="w-full border border-slate-200 rounded-lg p-2.5 focus:ring-2 focus:ring-emerald-500 outline-none" placeholder="e.g. Cafe Au Lait" />
                                    </div>

                                    <div className="space-y-1">
                                        <label className="text-sm font-medium text-slate-700">Plant Status</label>
                                        <select name="status" defaultValue={editingPlant?.status || 'active'} className="w-full border border-slate-200 rounded-lg p-2.5 focus:ring-2 focus:ring-emerald-500 outline-none bg-white">
                                            <option value="active">🌱 Healthy / Active</option>
                                            <option value="failed">❌ Did Not Grow (Failed)</option>
                                            <option value="skipped">🪨 Empty Space (Skipped)</option>
                                        </select>
                                    </div>

                                    <div className="space-y-1">
                                        <label className="text-sm font-medium text-slate-700 flex items-center justify-between">Additional Notes</label>
                                        <textarea name="notes" rows="2" defaultValue={editingPlant?.notes || ''} className="w-full border border-slate-200 rounded-lg p-2.5 focus:ring-2 focus:ring-emerald-500 outline-none resize-none"></textarea>
                                    </div>
                                    
                                    <div className="pt-2 flex gap-3">
                                        <button type="button" onClick={() => setIsManualModalOpen(false)} className="flex-1 px-4 py-2 border border-slate-200 text-slate-700 rounded-xl hover:bg-slate-50 transition">Cancel</button>
                                        <button type="submit" className="flex-1 px-4 py-2 bg-emerald-600 text-white rounded-xl hover:bg-emerald-700 transition font-medium">Save Details</button>
                                    </div>
                                </form>
                            </div>
                        </div>
                    )}

                    {viewingPhoto && (
                        <div className="fixed inset-0 bg-black/90 z-[60] flex items-center justify-center p-4 backdrop-blur-sm" onClick={() => setViewingPhoto(null)}>
                            <button onClick={() => setViewingPhoto(null)} className="absolute top-4 right-4 text-white bg-black/50 p-2 rounded-full hover:bg-white/20 transition">
                                <X className="w-6 h-6" />
                            </button>
                            <img src={viewingPhoto} alt="Dahlia" className="max-w-full max-h-[90vh] rounded-lg shadow-2xl object-contain animate-in zoom-in-95" onClick={(e) => e.stopPropagation()} />
                        </div>
                    )}
                </div>
            );
        }

        const root = createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
